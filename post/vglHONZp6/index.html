2020.11.18小测题解

发布于 2020 年 12 月 03 日 | 分类于 模拟赛题解 | 6分钟 | 1177字数

题面

思路

​ 最长链类的问题，我们已经学过的有，处理一个点为根的时候他的最长链，可以是最长链或者最长链加次长链，这个题里放到了一整棵树里面，我们直观的考虑到一个点上面的一段最长路径也可能被统计到答案中，我们需要知道的就是这一段，当前为根的最长，次长，将这三个里挑选两个取最大值。

​ 以根节点处理出当前子树内的最长链次长链比较好求，关键在于上面，是一个换根dp常用的技巧，也就是在dfs2的时候，我们的f，u已经不是仅仅在u子树内最长链和次长链了，而是对于u所有能到达的最长链和次长链，所以我们利用他从哪个儿子转移过最长链次长链来避免重复进行更新，最后就是最长链+次长链+本身就足够了，细节不少，但是整体思路过程比较好。

代码

#include <bits/stdc++.h>
using namespace std;
#define int long long
inline int read() {
    int r = 1, s = 0;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-')
        r = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        s = s * 10 + ch - '0';
        ch = getchar();
    }
    return r * s;
}

const int N = 1e6 + 10;

struct Edge{
    int nxt, to;
}e[N<<1];
int head[N], cnte;

//处理向上最长链，向下最长链，次长链
int n, q;
int a[N], f[N], g[N], s[N][2], lim;
//以x为根的最长链，次长链，最长链由哪个儿子转移，次长链由哪个儿子转移
int ans[N];


void add(int x, int y) {
    e[++ cnte] = (Edge){head[x], y};
    head[x] = cnte;
} 

void dfs(int u, int fath) {
    f[u] = g[u] = s[u][0] = s[u][1] = 0;
    for (int i = head[u];i;i = e[i].nxt) {
        int y = e[i].to;
        int z = a[y];
        if (y == fath) continue;
        dfs(y, u);
        if (a[y] > lim) continue;
        if (f[y] + z > f[u])
            g[u] = f[u], f[u] = f[y] + z, s[u][1] = s[u][0], s[u][0] = y;
        else if (f[y] + z > g[u]) 
                g[u] = f[y] + z, s[u][1] = y;
    }
} 

void dfs2(int u, int fath) {
    //o避免重复
	//如果当前点是最长链，那么我的值就是次长链
	//如果当前点是次长链，那么我的值就是最长链
    int o = s[fath][0] == u ? g[fath] : f[fath];
    if (a[fath] <= lim) {
        if (o + a[fath] > f[u])
            g[u] = f[u], f[u] = a[fath] + o, s[u][1] = s[u][0], s[u][0] = fath;
        else if (o + a[fath] > g[u])
                g[u] = o + a[fath], s[u][1] = fath;
    }
    if (a[u] <= lim && f[u] + g[u] + a[u] > 0)
    ans[u] = f[u] + g[u] + a[u];
    else ans[u] = 0;
    for (int i = head[u];i;i = e[i].nxt) {
        int y = e[i].to;
        if (y == fath) continue;
        dfs2(y, u);
    }
}

signed main() {
    n = read(), q = read();
    for (int i = 1;i <= n;++ i)
    a[i] = read();
    for (int i = 1, u, v;i < n;++ i) {
        u = read();
        v = read();
        add (u, v);
        add (v, u);
    }
    while (q --) {
        lim = read();
        dfs(1, 0);
        dfs2(1, 0);
        for (int i = 1;i <= n;++ i)
        printf("%lld ", ans[i]);
        puts(" ");
    }
    return 0;
}

题面

思路

树状数组优化dp题。

显然有O(n2)O(n^2)O(n2)暴力，设fif_ifi​为前i个人的最大价值，显然有

fi=max⁡(fi,fj+vali)(j+tj≤i,i−ti≥j)f_i = \max(f_i, f_j + val_i) (j+t_j \leq i,i-t_i \geq j) fi​=max(fi​,fj​+vali​)(j+tj​≤i,i−ti​≥j)

想要优化成O(n)O(n)O(n)就需要O(1)O(1)O(1)找出满足条件的转折点，我们可以先对第一条件进行排序，那么根据单调性第一个条件一定满足

第二个条件利用树状数组插入所有符合当前iii条件的因为排序过所以具有单调性，然后每次只查询满足第二个条件的最大值转移即可。

代码

#include <bits/stdc++.h>

using namespace std;

#define int long long
inline int read() {
    int r = 1, s = 0;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-')
        r = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        s = s * 10 + ch - '0';
        ch = getchar();
    }
    return r * s;
}

const int N = 1e6 + 10;
int n, a[N], b[N], w[N];
int f[N], tr[N<<1], P = 1;

struct Edge {
    int id, x;
    bool operator < (const Edge & a) const {
        return x < a.x;
    }
}m[N];

void change(int x, int v) {
    while (x <= n) {
        tr[x] = max(tr[x], v);
        x += (x & -x);
    }    
}

int get(int x) {
    int res = 0;
    while (x) {
        res = max(res, tr[x]);
        x -= (x & -x);
    }
    return res;
}


signed main() {
    n = read();
    for (int i = 1;i <= n;++ i)
    a[i] = read();
    for (int i = 1;i <= n;++ i) {
        b[i] = read(), w[i] = a[i] * b[i];
        m[i].id = i, m[i].x = i + a[i];
    }
    int ans = 0;
    sort(m + 1,m + 1 + n);
    for (int i = 1;i <= n;++ i) {
         while(P <= n && m[P].x <= i)
            change(m[P].id, w[m[P].id]), P ++;
        if (i - a[i] > 0) w[i] += get(i - a[i]);
        ans = max(ans, w[i]);
    }
    cout << ans;
}
